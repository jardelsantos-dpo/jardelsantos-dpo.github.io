    <script>
    function aplicarTema(tema) {
      document.documentElement.setAttribute('data-theme', tema);
      localStorage.setItem('portal-theme', tema);
    }

    function alternarTema() {
      const atual = document.documentElement.getAttribute('data-theme');
      const novo = atual === 'dark' ? 'light' : 'dark';
      aplicarTema(novo);

      const btnPainel = document.getElementById('btn-tema-painel');
      if (btnPainel) {
        btnPainel.textContent = novo === 'dark' ? '☀️ Claro' : '🌙 Escuro';
      }

      const btnIndex = document.getElementById('btn-tema-login');
      if (btnIndex) {
        btnIndex.textContent = novo === 'dark' ? '☀️ Claro' : '🌙 Escuro';
      }
    };

    window.painelLogout = function() {
      window.top.location.href = window.top.location.href;
    };

    window.painelAbrirModulo = function(modulo) {
      var container = document.getElementById('container-modulo');
      container.innerHTML = '<p>⏳ Carregando módulo...</p>';

      // Executa lógica comum para injetar e ler scripts de um módulo dinâmico
      function injetarHTMLModulo(html) {
        if (!html) {
          container.innerHTML = "<p style='color:red'>❌ Resposta vazia do servidor.</p>";
          return;
        }

        // 1. Limpa e insere o HTML básico
        container.innerHTML = html;

        // 2. Procura scripts e executa de forma segura
        var scripts = container.querySelectorAll('script');
        scripts.forEach(function(s) {
          try {
            if (s.src) {
              // Se for um script externo, carrega dinamicamente
              var scriptExterno = document.createElement('script');
              scriptExterno.src = s.src;
              document.head.appendChild(scriptExterno);
            } else {
              // Se for um script interno, usa eval para forçar a execução
              // Isso evita o erro de 'appendChild' no nó
              eval(s.textContent);
            }
          } catch (e) {
            console.error("Erro ao executar script do módulo:", e);
          }
        });
      }

      function manipularErro(erro) {
        container.innerHTML = "<p style='color:red'>❌ Erro: " + (erro.message || JSON.stringify(erro)) + "</p>";
      }

      // Roteamento dos Módulos ativos no Servidor
      if (modulo === 'meacao') {
        google.script.run
          .withSuccessHandler(injetarHTMLModulo)
          .withFailureHandler(manipularErro)
          .getCalculadoraMeacaoHTML();
      } 
      else if (modulo === 'multas') {
        // AJUSTE: Integração do novo módulo de multas do servidor
        google.script.run
          .withSuccessHandler(injetarHTMLModulo)
          .withFailureHandler(manipularErro)
          .getPainelMultasHTML();
      } 
      else {
        container.innerHTML = '<h3>Módulo <em>' + modulo + '</em> em desenvolvimento.</h3>';
      }
    };

    // Aplica o tema salvo ao carregar
    (function() {
      const tema = localStorage.getItem('portal-theme') || 'light';
      document.documentElement.setAttribute('data-theme', tema);

      const btnPainel = document.getElementById('btn-tema-painel');
      if (btnPainel) {
        btnPainel.textContent = tema === 'dark' ? '☀️ Claro' : '🌙 Escuro';
      }
    })();
    </script>
