    <div class="control-panel">
      <div class="control-group">

        <select id="meacao-select-mes"></select>

      <div class="control-group">
        <label for="meacao-select-ano">Ano Final</label>
        <select id="meacao-select-ano"></select>
      </div>
      <button class="btn-calcular" id="meacao-btn-calcular">Calcular</button>
    </div>

    <div id="meacao-resultado" class="loading">Selecione o período e clique em Calcular.</div>
  </div>

<script>
(function() {
  var dadosServidor = null;
  var activeIdx = "IPCA";
  var chart = null;
  var MESES = ["Janeiro","Fevereiro","Março","Abril","Maio","Junho","Julho","Agosto","Setembro","Outubro","Novembro","Dezembro"];

  // Popula os selects
  var selectMes = document.getElementById("meacao-select-mes");
  var selectAno = document.getElementById("meacao-select-ano");
  MESES.forEach(function(m, i) {
    var opt = document.createElement("option");
    opt.value = i; opt.textContent = m;
    selectMes.appendChild(opt);
  });
  for (var a = 2006; a <= 2030; a++) {
    var opt = document.createElement("option");
    opt.value = a; opt.textContent = a;
    selectAno.appendChild(opt);
  }
  var hoje = new Date();
  selectMes.value = hoje.getMonth();
  selectAno.value = hoje.getFullYear();

  // Botão calcular — chama o servidor
  document.getElementById("meacao-btn-calcular").addEventListener("click", function() {
    var mes = selectMes.value;
    var ano = selectAno.value;
    document.getElementById("meacao-resultado").innerHTML = "<div class='loading'>⏳ Calculando no servidor...</div>";

    google.script.run
      .withSuccessHandler(function(dados) {
        dadosServidor = dados;
        activeIdx = dados.indices[0];
        renderizarTudo();
      })
      .withFailureHandler(function(erro) {
        document.getElementById("meacao-resultado").innerHTML = "<div class='loading' style='color:red'>❌ Erro: " + (erro.message || JSON.stringify(erro)) + "</div>";
      })
      .obterDadosDashboard(mes, ano);
  });

  function fmt(v) {
    return "R$ " + v.toLocaleString("pt-BR", {minimumFractionDigits:2, maximumFractionDigits:2});
  }

  function renderizarTudo() {
    var d = dadosServidor;
    var r = d.resultados[activeIdx];

    var html = '';

    // Cards
    html += '<div class="cards">';
    html += '<div class="card"><div class="card-label">Saldo original (nov/2006)</div><div class="card-value">' + fmt(d.original) + '</div><div class="card-sub">CC + aplicação automática</div></div>';
    html += '<div class="card highlight"><div class="card-label">Total atualizado</div><div class="card-value" id="meacao-total-atualizado">' + fmt(r.total) + '</div><div class="card-sub" id="meacao-fator-label">Fator: ' + r.fator.toFixed(4) + '× (' + activeIdx + ')</div></div>';
    html += '<div class="card meacao-card"><div class="card-label">Meação ex-cônjuge (50%)</div><div class="card-value" id="meacao-valor">' + fmt(r.meacao) + '</div><div class="card-sub" id="meacao-indice-label">Índice: ' + activeIdx + '</div></div>';
    html += '<div class="card"><div class="card-label">Valorização</div><div class="card-value" id="meacao-valorizacao">' + (r.valorizacao >= 0 ? '+' : '') + r.valorizacao + '%</div><div class="card-sub" id="meacao-periodo-label">' + d.periodoLabel + '</div></div>';
    html += '</div>';

    // Seletor de índice
    html += '<div class="selector"><div class="selector-title">Selecione o índice de correção monetária</div><div class="btn-group" id="meacao-btn-group"></div><div class="idx-desc" id="meacao-idx-desc">' + d.resultados[activeIdx].desc + '</div></div>';

    // Composição
    html += '<div class="info-box"><div class="info-title">Composição do patrimônio partilhável</div><div class="info-grid">';
    html += '<div><div class="info-row"><span class="lbl">Saldo conta corrente</span><span class="val">' + fmt(d.ccOrig) + '</span></div><div class="info-row"><span class="lbl">Aplicação resgate automático</span><span class="val">' + fmt(d.aplicOrig) + '</span></div><div class="info-row"><span class="lbl">Total partilhável</span><span class="val">' + fmt(d.original) + '</span></div></div>';
    html += '<div><div class="info-row"><span class="lbl">CC atualizado</span><span class="val" id="meacao-cc-atualizado">' + fmt(r.cc) + '</span></div><div class="info-row"><span class="lbl">Aplicação atualizada</span><span class="val" id="meacao-aplic-atualizada">' + fmt(r.aplic) + '</span></div><div class="info-row"><span class="lbl">Total atualizado</span><span class="val" id="meacao-total-atualizado-2">' + fmt(r.total) + '</span></div></div>';
    html += '</div></div>';

    // Gráfico
    html += '<div class="chart-box"><div class="chart-title">Evolução da meação ao longo do tempo (R$)</div><canvas id="meacao-myChart" height="220"></canvas></div>';

    // Tabela comparativa
    html += '<div class="table-box"><div class="table-title" id="meacao-table-title">Comparativo entre todos os índices — ' + d.periodoLabel.split('→')[1].trim() + '</div>';
    html += '<table><thead><tr><th>Índice</th><th>Uso típico</th><th class="right">Fator</th><th class="right">Total atualizado</th><th class="right">Meação (50%)</th></tr></thead><tbody id="meacao-comp-tbody"></tbody></table>';
    html += '<p class="note">Valores calculados no servidor com base nas séries históricas reais por interpolação linear entre marcos contábeis anuais.</p></div>';

    document.getElementById("meacao-resultado").innerHTML = html;

    // Botões de índice
    var btnGroup = document.getElementById("meacao-btn-group");
    d.indices.forEach(function(idx) {
      var b = document.createElement("button");
      b.className = "btn-idx" + (idx === activeIdx ? " active" : "");
      b.textContent = idx;
      b.addEventListener("click", function() { trocarIndice(idx); });
      btnGroup.appendChild(b);
    });

    // Tabela
    buildTable();

    // Gráfico
    buildChart();
  }

  function trocarIndice(idx) {
    activeIdx = idx;
    var d = dadosServidor;
    var r = d.resultados[idx];
    document.getElementById("meacao-total-atualizado").textContent   = fmt(r.total);
    document.getElementById("meacao-fator-label").textContent        = "Fator: " + r.fator.toFixed(4) + "× (" + idx + ")";
    document.getElementById("meacao-valor").textContent              = fmt(r.meacao);
    document.getElementById("meacao-indice-label").textContent       = "Índice: " + idx;
    document.getElementById("meacao-valorizacao").textContent        = (r.valorizacao >= 0 ? "+" : "") + r.valorizacao + "%";
    document.getElementById("meacao-cc-atualizado").textContent      = fmt(r.cc);
    document.getElementById("meacao-aplic-atualizada").textContent   = fmt(r.aplic);
    document.getElementById("meacao-total-atualizado-2").textContent = fmt(r.total);
    document.getElementById("meacao-idx-desc").textContent           = r.desc;
    document.querySelectorAll(".btn-idx").forEach(function(b) { b.classList.toggle("active", b.textContent === idx); });
    atualizarGrafico();
  }

  function buildTable() {
    var tbody = document.getElementById("meacao-comp-tbody");
    tbody.innerHTML = "";
    var d = dadosServidor;
    d.indices.forEach(function(idx) {
      var r = d.resultados[idx];
      var tr = document.createElement("tr");
      tr.innerHTML = "<td><strong>" + idx + "</strong></td><td><span class='badge " + r.badge + "'>" + r.uso + "</span></td><td class='right'>" + r.fator.toFixed(4) + "×</td><td class='right'>" + fmt(r.total) + "</td><td class='right'>" + fmt(r.meacao) + "</td>";
      tbody.appendChild(tr);
    });
    var trOrig = document.createElement("tr");
    trOrig.innerHTML = "<td>Original (sem correção)</td><td></td><td class='right'>1,0000×</td><td class='right'>" + fmt(d.original) + "</td><td class='right'>" + fmt(d.original/2) + "</td>";
    tbody.appendChild(trOrig);
  }

  function buildChart() {
    var ctx = document.getElementById("meacao-myChart").getContext("2d");
    var d = dadosServidor;
    var COLORS = {"IPCA":"#378ADD","INPC":"#1D9E75","IGP-M":"#EF9F27","IGP-DI":"#D85A30","SELIC":"#E24B4A"};

    var datasets = d.indices.map(function(idx) {
      return {
        label: idx,
        data: d.grafico.datasets[idx].data,
        borderColor: COLORS[idx],
        backgroundColor: "transparent",
        borderWidth: idx === activeIdx ? 2.5 : 1,
        borderDash: idx === activeIdx ? [] : [4,3],
        pointRadius: idx === activeIdx ? 3 : 0,
        pointHoverRadius: 5,
        tension: 0.1
      };
    });

    if (chart) chart.destroy();
    chart = new Chart(ctx, {
      type: "line",
      data: { labels: d.grafico.labels, datasets: datasets },
      options: {
        responsive: true,
        interaction: { mode: "index", intersect: false },
        plugins: {
          legend: { position: "bottom", labels: { font: { size: 12 }, usePointStyle: true, boxWidth: 20, padding: 16 } },
          tooltip: { callbacks: { label: function(c) { return " " + c.dataset.label + ": " + fmt(c.parsed.y); } } }
        },
        scales: {
          x: { grid: { color: "rgba(0,0,0,0.05)" }, ticks: { font: { size: 11 } } },
          y: { grid: { color: "rgba(0,0,0,0.05)" }, ticks: { font: { size: 11 }, callback: function(v) { return "R$ " + (v/1000).toFixed(0) + "k"; } } }
        }
      }
    });
  }

  function atualizarGrafico() {
    if (!chart) return;
    var COLORS = {"IPCA":"#378ADD","INPC":"#1D9E75","IGP-M":"#EF9F27","IGP-DI":"#D85A30","SELIC":"#E24B4A"};
    chart.data.datasets.forEach(function(ds) {
      ds.borderWidth = ds.label === activeIdx ? 2.5 : 1;
      ds.borderDash  = ds.label === activeIdx ? [] : [4,3];
      ds.pointRadius = ds.label === activeIdx ? 3 : 0;
    });
    chart.update();
  }

})();
</script>
