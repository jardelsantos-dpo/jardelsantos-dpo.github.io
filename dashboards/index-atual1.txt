    let tokenSessao = "";
    let usuarioTemporario = "";
    let emailMFAEmAndamento = "";
    let cooldownReenvio = false;

    window.addEventListener('load', function() {
      const params = new URLSearchParams(window.location.search);
      const token = params.get('token');
      if (token && token.trim() !== "") {
        validarLinkRecuperacao(token.trim());
      }
    });

    // ==========================================
    // LÓGICA DE LOGIN DINÂMICA
    // ==========================================
    function fazerLogin() {
      const user = document.getElementById('usuario').value;
      const pass = document.getElementById('senha').value;
      const btn = document.getElementById('btnEntrar');
      
      btn.className = "btn-submit";
      btn.innerText = "Verificando...";
      btn.disabled = true;
      google.script.run
      .withSuccessHandler(function(r) {
          if (r && r.sucesso) {
            if (r.mfaRequerido) {
              emailMFAEmAndamento = document.getElementById('usuario').value.trim();
              
              let botaoMFA = document.getElementById('btnValidarMFA');
              if (botaoMFA) {
                botaoMFA.className = "btn-submit";
                botaoMFA.innerText = "Confirmar e Acessar";
                botaoMFA.disabled = false;
              }
              document.getElementById('codigoMFA').value = ""; 

              // =====================================================
              // ORQUESTRAÇÃO DE UX DA TELA DE MFA
              // =====================================================
              const tituloMFA = document.querySelector('#boxMFA h2');
              const subtextoMFA = document.getElementById('subtextoMFA');
              const containerReenvio = document.getElementById('containerReenvioMFA');
              const btnIrParaSetup = document.getElementById('btnIrParaSetupMFA');
              const btnFallbackEmail = document.getElementById('btnFallbackEmail');

              if (r.tipoMFA === "APP") {
                if (tituloMFA) tituloMFA.innerText = "App Authenticator";
                if (subtextoMFA) subtextoMFA.innerText = "Abra o seu aplicativo autenticador e digite o código gerado.";
                
                containerReenvio.style.display = "none";
                btnIrParaSetup.classList.add('hidden');
                btnFallbackEmail.classList.remove('hidden'); // Exibe botão "Problemas? Enviar e-mail"
              } else {
                if (tituloMFA) tituloMFA.innerText = "Verificação de Segurança";
                if (subtextoMFA) subtextoMFA.innerText = "Insira o código de verificação enviado para o seu e-mail cadastrado.";
                
                containerReenvio.style.display = "block";
                btnIrParaSetup.classList.remove('hidden'); // Exibe botão de Setup
                btnFallbackEmail.classList.add('hidden');
              }

              document.getElementById('boxLogin').classList.add('hidden');
              document.getElementById('boxMFA').classList.remove('hidden');
          } else if (r.precisaTrocarSenha) {
              usuarioTemporario = document.getElementById('usuario').value.trim();
              document.getElementById('boxLogin').classList.add('hidden');
              document.getElementById('boxPreDefinido').classList.remove('hidden');
          }
          } else {
            const msgErro = (r && r.mensagem) ? r.mensagem : "Usuário ou senha incorretos.";
            btn.className = "btn-submit btn-erro";
            btn.innerText = msgErro;
            btn.disabled = false;
            setTimeout(() => {
              btn.className = "btn-submit";
              btn.innerText = "Acessar Portal";
            }, 4000);
          }
        })
        .withFailureHandler(function(err) {
          btn.className = "btn-submit btn-erro";
          btn.innerText = "Erro no servidor. Tente novamente.";
          btn.disabled = false;
          setTimeout(() => { btn.className = "btn-submit"; btn.innerText = "Acessar Portal"; }, 4000);
        })
        .loginUsuario(user, pass);
    }

    function reenviarMFA() {
      if (cooldownReenvio) return;
      const btn = document.getElementById('btnReenviarMFA');
      const msg = document.getElementById('msgReenvioMFA');

      if (!emailMFAEmAndamento) return;

      btn.disabled = true;
      btn.innerText = "Enviando...";
      msg.style.color = "var(--txt-secundario)"; msg.innerText = "";

      google.script.run
        .withSuccessHandler(function(r) {
          if (r && r.sucesso) {
            msg.style.color = "var(--sucesso)"; msg.innerText = "✓ " + r.mensagem;
            document.getElementById('codigoMFA').value = "";
            
            cooldownReenvio = true;
            let segundos = 60;
            btn.innerText = `Aguarde ${segundos}s`;
            const intervalo = setInterval(() => {
              segundos--;
              btn.innerText = `Aguarde ${segundos}s`;
              if (segundos <= 0) {
                clearInterval(intervalo); cooldownReenvio = false; btn.disabled = false;
                btn.innerText = "🔁 Reenviar código"; msg.innerText = "";
              }
            }, 1000);
          } else {
            msg.style.color = "var(--perigo)"; msg.innerText = (r && r.mensagem) ? r.mensagem : "Erro.";
            btn.disabled = false; btn.innerText = "🔁 Reenviar código";
          }
        })
        .withFailureHandler(function() {
          msg.style.color = "var(--perigo)"; msg.innerText = "Erro de rede.";
          btn.disabled = false; btn.innerText = "🔁 Reenviar código";
        })
        .reenviarCodigoMFA(emailMFAEmAndamento);
    }

    function alterarSenhaProvisoria() {
      const nSenha = document.getElementById('novaSenha').value;
      const cSenha = document.getElementById('confirmaSenha').value;
      const btn = document.getElementById('btnMudarSenha');
      
      // 1. Validação de Coincidência
      if(nSenha !== cSenha) {
        exibirErroBotao(btn, "As senhas não coincidem!");
        return;
      }

      // 2. Validação de Complexidade (8+ carac, Maiúscula, Minúscula, Número, Especial)
      const regexSeguranca = /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[@$!%*?&])[A-Za-z\d@$!%*?&]{8,}$/;
      if(!regexSeguranca.test(nSenha)) {
        exibirErroBotao(btn, "Senha fraca! Use: 8+ carac, Maiúsc, Minúsc, Número e Especial (@$!%*?&)");
        return;
      }

      // 3. Envio ao Servidor
      btn.className = "btn-submit";
      btn.innerText = "Salvando credenciais...";
      btn.disabled = true;

      google.script.run
        .withSuccessHandler(function(r) {
          if(r && r.sucesso) {
            btn.className = "btn-submit btn-sucesso";
            btn.innerText = "Senha Salva! Entrando...";
            tokenSessao = r.token;
            setTimeout(() => {
              document.getElementById('boxPreDefinido').classList.add('hidden');
              entrarNoPortal();
            }, 1200);
          } else {
            const msg = (r && r.mensagem) ? r.mensagem : "Erro ao atualizar.";
            exibirErroBotao(btn, msg);
          }
        })
        .withFailureHandler(function(err) {
          exibirErroBotao(btn, "Erro de comunicação.");
        })
        .atualizarSenha(usuarioTemporario, nSenha);
    }

    function alternarTema() {
      const htmlElement = document.documentElement;
      let novoTema = htmlElement.getAttribute('data-theme') !== 'dark' ? 'dark' : 'light';
      htmlElement.setAttribute('data-theme', novoTema);
      localStorage.setItem('temaPortal', novoTema);
      
      const btnPainel = document.getElementById('btnTrocaTemaPainel');
      if (btnPainel) { 
        btnPainel.innerText = novoTema === 'dark' ? "☀️" : "🌙"; 
        
        // Aplica o efeito de rotação
        btnPainel.style.transform = "rotate(360deg)";
        
        // Remove a rotação após a animação (200ms) para estar pronto para o próximo clique
        setTimeout(() => { btnPainel.style.transform = "none"; }, 200);
      }
      
      const btnLogin = document.getElementById('topoThemeBtn');
      if (btnLogin) { 
        btnLogin.innerHTML = novoTema === 'dark' ? '<span class="theme-icon">☀️</span> <span class="theme-text">Modo Claro</span>' : '<span class="theme-icon">🌙</span> <span class="theme-text">Modo Escuro</span>'; 
      }
    }

    function entrarNoPortal() {
      document.getElementById('boxMFA').classList.add('hidden');
      document.getElementById('boxPreDefinido').classList.add('hidden');
      document.getElementById('boxLogin').classList.add('hidden');
      document.getElementById('boxPortal').classList.remove('hidden');
      document.getElementById('topoThemeBtn').classList.add('hidden');
      
      google.script.run
        .withSuccessHandler(function(r) {
          const area = document.getElementById('itensArea'); area.innerHTML = "";
          if(r.sucesso && r.arquivos.length > 0) {
            r.arquivos.forEach(arq => {
              let div = document.createElement('div'); div.className = "item-arq"; div.id = "card-" + arq.id; 
              
              const partesNome = arq.nome.split('.');
              const extensao = partesNome.length > 1 ? partesNome.pop().toLowerCase() : '';
              const nomeSemExtensao = partesNome.join('.');
              
              let BadgeTexto = extensao.toUpperCase(); let classeBadge = "badge-generic"; let emojiIcone = "📄";
              switch (extensao) {
                case 'pdf': classeBadge = "badge-pdf"; emojiIcone = "📕"; break;
                case 'docx': case 'doc': BadgeTexto = "DOCX"; classeBadge = "badge-word"; emojiIcone = "📘"; break;
                case 'xlsx': case 'xls': case 'csv': BadgeTexto = "XLSX"; classeBadge = "badge-excel"; emojiIcone = "📗"; break;
                case 'txt': classeBadge = "badge-txt"; emojiIcone = "📝"; break;
              }

              let botoes = `<button class="btn-view" onclick="visualizar('${arq.id}', '${arq.nome}')">Visualizar</button>`;
              if(r.perfil === 'Advogado' && arq.downloadUrl) { botoes += `<button class="btn-down" onclick="registrarBaixar('${arq.nome}', '${arq.downloadUrl}')">Baixar</button>`; }
              
              //div.innerHTML = `<span class="item-nome">${emojiIcone} ${nomeSemExtensao} <span class="badge-tipo ${classeBadge}">${BadgeTexto}</span></span> <div class="item-acoes">${botoes}</div>`;
              div.innerHTML = `
                <span class="item-nome">
                  <span class="item-nome-texto">${emojiIcone} ${nomeSemExtensao}</span>
                  <span class="badge-tipo ${classeBadge}">${BadgeTexto}</span>
                </span> 
                <div class="item-acoes">${botoes}</div>
              `;
              area.appendChild(div);
              
            });
          } else { area.innerHTML = "Nenhum arquivo disponível."; }
        }).listarDocumentos(tokenSessao);
    }
